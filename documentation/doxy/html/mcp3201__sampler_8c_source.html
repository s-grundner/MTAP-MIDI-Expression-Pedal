29static void IRAM_ATTR sampler_task(void *arg)
30{
31 mcp3201_sampler_t *sampler = (mcp3201_sampler_t *)arg;
32 for (;;)
33 {
34 spi_event_t evt;
35 if (xQueueReceive(sampler->dma_queue, &evt, portMAX_DELAY) == pdTRUE)
36 {
37 if (evt == SPI_EV_SEND_DMA_READY)
38 {
39 size_t bytes_read = 0;
40 do
41 {
42 // fill audio buffer
43 size_t bytes_to_read = RESAMPLE_DENOMINATOR * (sampler->buffer_size - sampler->buffer_pos);
44 void *buffer_position = (void *)(sampler->buffer + sampler->buffer_pos);
45
46 // read data from spi
47 spi_transaction_t t = {
48 .user = (void *)sampler->mcp_handle,
49 .length = bytes_to_read,
50 .rx_buffer = buffer_position,
51 };
52 sampler->mcp_handle->ongoing_transaction = &t;
53 spi_device_get_trans_result(
54 sampler->mcp_handle->spi,
55 (spi_transaction_t **)(sampler->mcp_handle->ongoing_transaction),
56 READER_TIMEOUT_TICKS);
57 bytes_read = t.rxlength;
58
59 sampler->buffer_pos += bytes_read / RESAMPLE_DENOMINATOR;
60
61 if (sampler->buffer_pos == sampler->buffer_size)
62 {
63 // send data to DSP queue
64 sampler->buffer_pos = 0;
65 xQueueSend(sampler->dsp_queue, &sampler->buffer, portMAX_DELAY);
66 }
67 } while (bytes_read > 0);
68 }
69 }
70 }
71}

85{
86 if (mcp_handle == NULL || recv_queue == NULL || buffer_size == 0 || f_sample == 0)
87 {
88 ESP_LOGE(TAG, "Invalid arguments");

90 }
91
92 mcp3201_sampler_t *sampler = (mcp3201_sampler_t *)malloc(sizeof(mcp3201_sampler_t));
93 *sampler = (mcp3201_sampler_t){
94 .mcp_handle = mcp_handle,
95 .buffer_size = buffer_size,
96 .buffer = (size_t *)malloc(sizeof(size_t) * buffer_size),
97 .buffer_pos = 0,
98 .dma_queue = xQueueCreate(4, sizeof(spi_event_t)),
99 .dsp_queue = recv_queue};
100
101 while (spi_device_acquire_bus(mcp_handle->spi, portMAX_DELAY) != ESP_OK)
102 ESP_LOGE(TAG, "Failed to acquire bus\n retrying...");
103
104 xTaskCreatePinnedToCore(sampler_task, "sampler_task", RESAMPLE_DENOMINATOR * buffer_size, sampler, 5, &sampler_task_handle, 0);
105 xTaskCreatePinnedToCore(reader_task, "reader_task", 2048, sampler, 5, &reader_task_handle, 0);
106

108}
