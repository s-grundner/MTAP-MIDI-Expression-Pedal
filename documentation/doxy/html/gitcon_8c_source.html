43static void dsp_task(void *arg)
44{
45 // gitcon driver context handler
46 gitcon_handle_t gitcon_handle = (gitcon_handle_t)arg;
47
48 // audio buffer variables
49 uint16_t *audio_buffer = NULL;
50 float *audio_buffer_float = (float *)malloc(FFT_SIZE * sizeof(float));
51 if (audio_buffer_float == NULL)
52 {
53 ESP_LOGE(TAG, "Could not allocate memory for audio_buffer_float");
54 gitcon_exit(gitcon_handle);
55 return;
56 }
57
58 // fft variables
59 float fft_buffer[FFT_SIZE];
60 float magnitude[FFT_SIZE / 2];
61 float frequency[FFT_SIZE / 2];
62 unsigned char keyNR[FFT_SIZE / 2];
63 float ratio = (float)F_SAMPLE_HZ / (float)FFT_SIZE;
64
65 // active notes resulting from fft
66 midi_message_t *active_notes = (midi_message_t *)malloc(128 * sizeof(midi_message_t));
67 if (active_notes == NULL)
68 {
69 ESP_LOGE(TAG, "Could not allocate memory for active_notes");
70 gitcon_exit(gitcon_handle);
71 return;
72 }
73
74 // initialize active_notes
75 for (int i = 0; i < 128; i++)
76 {
77 active_notes[i].channel = 0;
78 active_notes[i].status = MIDI_STATUS_NOTE_OFF;
79 active_notes[i].param1 = i;
80 active_notes[i].param2 = 0;
81 }
82
83 // window counter to sweep through the audio_buffer_float
84 char window_counter = 0;
85
86 for (;;)
87 {
88 // delay to avoid overflow and to allow other tasks to run
89 vTaskDelay(10 / portTICK_PERIOD_MS);
90
91 // get audio buffer from sampler
92 if (xQueueReceive(gitcon_handle->sampler->dsp_queue, &audio_buffer, portMAX_DELAY) == pdFALSE)
93 continue; // skip iteration if queue is empty
94
97 float *start_pos = audio_buffer_float;
98 audio_buffer_float += (AUDIO_BUFFER_SIZE * window_counter); // move pointer to the next window
99 window_counter = (window_counter + 1) % (FFT_WINDOW_SIZE); // increment window counter
100
101 // starting in a new window and fill the buffer with the new data
102 for (int i = 0; i < AUDIO_BUFFER_SIZE; i++)
103 audio_buffer_float[i] = UINT16_TO_FLOAT(audio_buffer[i]);
104 audio_buffer_float = start_pos;
105
106#ifdef DEBUG_BETTER_SERIAL_PLOTTER
107 for (int i = 0; i < FFT_SIZE; i++)
108 printf("%f\n", audio_buffer_float[i]); // for debugging in BetterSerialPlotter
109#endif
111
113 fft_config_t *fft_plan = fft_init(FFT_SIZE, FFT_REAL, FFT_FORWARD, audio_buffer_float, fft_buffer);
114 if (fft_plan == NULL)
115 {
116 ESP_LOGE(TAG, "FFT plan could not be created");
117 vTaskDelay(1000 / portTICK_PERIOD_MS);
118 continue;
119 }
120 fft_execute(fft_plan);
121
122 bool invalid_key = false; // invalid key flag
123 for (int k = 1; k < FFT_SIZE / 2; k++)
124 {
125 // detect fundamental frequencies
126 frequency[k] = k * ratio;
127 // convert to note number on piano roll
128 keyNR[k] = (unsigned char)round(log2(frequency[k] / CONCERT_A) * 12 + CONCERT_A_NOTE) % 128;
129 // calculate magnitude (absolute value of complex number)
130 magnitude[k] = 2 * sqrt(pow(fft_buffer[2 * k], 2) + pow(fft_buffer[2 * k + 1], 2)) / FFT_SIZE;
131 }
132
133 // calculate max magnitude for thresholding
134 float max = 0;
135 for (int i = 0; i < FFT_SIZE / 2; i++)
136 max = (magnitude[i] > max) ? magnitude[i] : max;
137
141 if (max < 0.0005)
142 max = 100;
143
144 // check if magnitudes pass a certain threshold
145 for (int k = 1; k < FFT_SIZE / 2; k++)
146 {
147 // switch off notes that are not active
148 if ((magnitude[k] < max * SENSITIVITY) || MIDI_KEY_BOUNDARY(keyNR[k]))
149 {
150 active_notes[keyNR[k]].status = MIDI_STATUS_NOTE_OFF;
151 continue; // skip to next iteration threshold is not passed
152 }
153#ifdef DEBUG_DSP
154 ESP_LOGI(TAG, "keyNR: %d, magnitude: %f, frequency: %f", keyNR[k], magnitude[k], frequency[k]);
155#endif
156 active_notes[keyNR[k]].status = MIDI_STATUS_NOTE_ON;
157 active_notes[keyNR[k]].param2 = (uint8_t)(magnitude[k] / max * 127);
158 }
159 // send saved notes to MIDI queue
160 xQueueSend(gitcon_handle->midi_queue, &active_notes, portMAX_DELAY);
161 fft_destroy(fft_plan);
162 } // for(;;)
163} // dsp_task

165static void midi_task(void *arg)
166{
167 // gitcon driver context handler
168 gitcon_handle_t gitcon_handle = (gitcon_handle_t)arg;
169 midi_message_t *active_notes = NULL;
170 midi_status_t previous_states[128] = {0};
171 for (;;)
172 {
174 vTaskDelay(10 / portTICK_PERIOD_MS);
175
176 if (xQueueReceive(gitcon_handle->midi_queue, &active_notes, portMAX_DELAY) == pdFALSE)
177 continue; // skip iteration if queue is empty
178
180 // send MIDI messages to UART
181 for (size_t i = 0; i < 128; i++)
182 {
183 if (active_notes[i].status == previous_states[i])
184 continue; // continue if note has not changed
185 // send message to MIDI UART
186 ESP_ERROR_CHECK(midi_write(gitcon_handle->midi_handle, &active_notes[i]));
187 previous_states[i] = active_notes[i].status;
188 }
189 } // for(;;)
190} // midi_task

196esp_err_t gitcon_init(gitcon_context_t **out_handle)
197{
198 gitcon_context_t *gitcon_cfg = (gitcon_context_t *)malloc(sizeof(gitcon_context_t));
199 if (!gitcon_cfg)
200 return ESP_ERR_NO_MEM;
201
202 // create queue for audio data (passed into sampler)
203 QueueHandle_t dsp_queue = xQueueCreate(10, sizeof(size_t *));
204
205 // create queue for midi messages
206 gitcon_cfg->midi_queue = xQueueCreate(5, sizeof(midi_handle_t *));
207 if (!gitcon_cfg->midi_queue)
208 return ESP_ERR_NO_MEM;
209
210#ifdef USE_MCP3201
211 // ------------------------------------------------------------
212 // SPI
213 // ------------------------------------------------------------
214 spi_bus_config_t bus_cfg = {
215 .miso_io_num = SPI_MISO,
216 .mosi_io_num = SPI_MOSI,
217 .sclk_io_num = SPI_SCLK,
218 .max_transfer_sz = 32,
219 };
220 spi_bus_initialize(SPI_DEV, &bus_cfg, DMA_CHAN);
221
222 // ------------------------------------------------------------
223 // MCP3201 (ADC)
224 // ------------------------------------------------------------
225 mcp3201_handle_t mcp_handle;
226 mcp3201_config_t mcp_cfg = {
227 .host = SPI_DEV,
228 .cs_io = SPI_CS,
229 .miso_io = SPI_MISO,
230 .mosi_io = SPI_MOSI};
231 // initialize ADC and store in gitcon handle
232 ESP_ERROR_CHECK(mcp3201_init(&mcp_handle, &mcp_cfg));
233 gitcon_cfg->sampler = mcp3201_sampler_start(mcp_handle, dsp_queue, AUDIO_BUFFER_SIZE, F_SAMPLE_HZ);
234#else
235 gitcon_cfg->sampler = i2s_sampler_start(INTERNAL_ADC_CHANNEL, dsp_queue, AUDIO_BUFFER_SIZE, F_SAMPLE_HZ);
236#endif
237
238 // ------------------------------------------------------------
239 // MIDI
240 // ------------------------------------------------------------
241
242 // Configure a handle parameter for MIDI
243 midi_handle_t midi_handle;
244 midi_config_t midi_cfg = {
245 .uart_num = MIDI_UART,
246 .baudrate = MIDI_BAUD,
247 .rx_io = MIDI_RX,
248 .tx_io = MIDI_TX};
249 // Initialize MIDI and store in gitcon handle
250 ESP_ERROR_CHECK(midi_init(&midi_handle, &midi_cfg));
251 gitcon_cfg->midi_handle = midi_handle;
252
253 // ------------------------------------------------------------
254 // INIT RTOS
255 // ------------------------------------------------------------
256
257 ESP_LOGI(TAG, "Creating RTOS tasks...");
259 if (xTaskCreatePinnedToCore(dsp_task, "dsp_task", 1 << 16, gitcon_cfg, 5, &dsp_task_handle, 1) == pdFALSE)
260 return ESP_ERR_NO_MEM;
261
263 if (xTaskCreatePinnedToCore(midi_task, "midi_task", 2048, gitcon_cfg, 5, &midi_task_handle, 0) == pdFALSE)
264 return ESP_ERR_NO_MEM;
265
266 // Pass final configuration to outer parameters
267 *out_handle = gitcon_cfg;
268 return ESP_OK;
269}

272{
273 ESP_ERROR_CHECK(midi_exit(handle->midi_handle));
274
275 // stop tasks
276 vTaskDelete(dsp_task_handle);
277 vTaskDelete(midi_task_handle);
278

285#endif
286

288 return ESP_OK;
