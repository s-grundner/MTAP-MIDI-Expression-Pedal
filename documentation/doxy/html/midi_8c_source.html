92esp_err_t midi_write(midi_handle_t handle, midi_message_t *msg)
93{
94 int len = 0;
95 const char data[] = {msg->status | msg->channel, msg->param1, msg->param2};
96
97 // switch status to determine message length
98 switch (msg->status)
99 {
100 case MIDI_STATUS_NOTE_OFF:
101 case MIDI_STATUS_NOTE_ON:
102 case MIDI_STATUS_CONTROL_CHANGE:
103 case MIDI_STATUS_PITCH_BEND:
104 case MIDI_STATUS_POLYPHONIC_KEY_PRESSURE:
105 len = uart_write_bytes(handle->cfg.uart_num, data, MIDI_BYTE_SIZE_DEFAULT);
106 break;
107 case MIDI_STATUS_PROGRAM_CHANGE:
108 case MIDI_STATUS_CHANNEL_PRESSURE:
109 len = uart_write_bytes(handle->cfg.uart_num, data, MIDI_BYTE_SIZE_SHORT);
110 break;
111 default:
112 ESP_LOGE(TAG, "midi_send: invalid status: %02X", msg->status);
113 return ESP_ERR_INVALID_ARG;
114 break;
115 }
116 switch (len)
117 {
118 case -1:
119 ESP_LOGE(TAG, "uart_write_bytes failed");
120 return ESP_FAIL;
121 break;
122 case 0:
123 ESP_LOGE(TAG, "uart_write_bytes timed out");
124 return ESP_ERR_TIMEOUT;
125 break;
126 default:

128 break;
129 }
